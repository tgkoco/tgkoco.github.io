#!/bin/bash
#whiptail --title "<dialog box title>" --yesno "<text to show>" <height> <width>
if (whiptail --title "Yes/No 对话框" --yesno "选择yes或者no." 10 60);
then echo "你选择的yes$?";
else echo "你选择的no$?";
fi
#!/bin/bash
if (whiptail --title "选择对话框" --yes-button "男" --no-button "女"  --yesno "选择你的性别？" 10 60)
then echo "你选择的是男 $?."
else echo "你选择的是女 $?."
fi

表单输入框

#!/bin/bash
#whiptail --title "<input box title>" --inputbox "<text to show>" <height> <width> <default-text>
NAME=$(whiptail --title "表单输入框" --inputbox "你的名称?" 10 60 "koco" 3>&1 1>&2 2>&3)

exitstatus=$?
if [ $exitstatus = 0 ];
then echo "你的名字是:" $NAME
else echo "你选择退出."
fi

进度条

#!/bin/bash
#whiptail --gauge "<test to show>" <height> <width> <inital percent>
{
for ((i = 0 ; i <= 100 ; i+=10)); 
do
    sleep 0.1
    echo $i
done
} | whiptail --gauge "请等待" 6 60 0
